# 訓練數據分類與管理系統

> 區分自動訓練數據與玩家參與數據，提升數據價值和分析準確性

---

## 🎯 核心目標

### 數據分類
1. **自動訓練數據** (AI vs AI)
   - 用於模型初步訓練
   - 大量生成，快速迭代
   - 質量穩定但創新性較低

2. **玩家參與數據** (Human vs AI) ⭐
   - 用於評估 AI 表現
   - 真實人類策略和反應
   - 高價值，用於改進 AI
   - 可用於研究人機交互

## 🗄️ 數據庫架構設計

### 1. 訓練會話表 (training_sessions)

```sql
CREATE TABLE training_sessions (
    id TEXT PRIMARY KEY,                    -- UUID
    session_type TEXT NOT NULL,             -- 'auto_training' or 'player_mode'
    mode TEXT,                              -- 'fast', 'demo', 'player_as_expert', 'player_as_scammer'
    
    -- 基本信息
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    status TEXT DEFAULT 'in_progress',      -- 'in_progress', 'completed', 'cancelled'
    
    -- 配置
    victim_persona TEXT,                    -- 'elderly', 'average', 'overconfident'
    scam_tactic TEXT,                       -- 詐騙類型
    
    -- 玩家信息（僅 player_mode）
    player_id TEXT,                         -- 玩家標識
    player_role TEXT,                       -- 'expert' or 'scammer'
    player_score INTEGER,                   -- 玩家得分
    
    -- 結果
    outcome TEXT,                           -- 'SUCCESS', 'FAILURE', 'PARTIAL_SUCCESS'
    round_count INTEGER,                    -- 對話輪數
    
    -- 評分（來自 RecorderAgent）
    scammer_score INTEGER,
    expert_score INTEGER,
    
    -- 元數據
    metadata TEXT,                          -- JSON: 其他信息
    
    -- 索引
    INDEX idx_session_type (session_type),
    INDEX idx_player_id (player_id),
    INDEX idx_created_at (created_at)
);
```

### 2. 對話記錄表 (conversation_messages)

```sql
CREATE TABLE conversation_messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id TEXT NOT NULL,
    
    -- 消息信息
    round_number INTEGER,
    speaker TEXT,                           -- 'scammer', 'victim', 'expert', 'player'
    message TEXT NOT NULL,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- 來源標記
    is_player_generated BOOLEAN DEFAULT FALSE,  -- 是否由玩家生成
    
    -- 上下文
    trust_in_scammer INTEGER,
    trust_in_expert INTEGER,
    
    -- 性能指標（如果可用）
    metrics TEXT,                           -- JSON: 性能指標
    
    FOREIGN KEY (session_id) REFERENCES training_sessions(id)
);
```

```python
@router.get("/api/data/analysis")
async def analyze_data(
    data_type: str = "all",  # 'all', 'auto_training', 'player_mode'
    start_date: Optional[str] = None,
    end_date: Optional[str] = None
):
    """分析訓練數據"""
    conn = sqlite3.connect(DATABASE_PATH)
    
    # 構建查詢
    query = """
        SELECT 
            session_type,
            COUNT(*) as session_count,
            AVG(scammer_score) as avg_scammer_score,
            AVG(expert_score) as avg_expert_score,
            AVG(round_count) as avg_rounds,
            SUM(CASE WHEN outcome = 'SUCCESS' THEN 1 ELSE 0 END) as success_count
        FROM training_sessions
        WHERE 1=1
    """
    
    params = []
    
    if data_type != "all":
        query += " AND session_type = ?"
        params.append(data_type)
    
    if start_date:
        query += " AND created_at >= ?"
        params.append(start_date)
    
    if end_date:
        query += " AND created_at <= ?"
        params.append(end_date)
    
    query += " GROUP BY session_type"
    
    cursor = conn.cursor()
    cursor.execute(query, params)
    results = cursor.fetchall()
    conn.close()
    
    return {
        "data": [
            {
                "type": row[0],
                "session_count": row[1],
                "avg_scammer_score": row[2],
                "avg_expert_score": row[3],
                "avg_rounds": row[4],
                "success_count": row[5]
            }
            for row in results
        ]
    }
```

```python
@router.post("/api/data/export")
async def export_data(request: ExportDataRequest):
    """導出訓練數據用於 Fine-tuning"""
    conn = sqlite3.connect(DATABASE_PATH)
    cursor = conn.cursor()
    
    # 根據類型篩選 sessions
    query = """
        SELECT id FROM training_sessions
        WHERE session_type = ?
        AND status = 'completed'
        AND scammer_score >= ?  -- 質量門檻
        AND expert_score >= ?
    """
    
    cursor.execute(query, (
        request.data_type,  # 'auto_training' or 'player_mode'
        request.min_scammer_score or 50,
        request.min_expert_score or 50
    ))
    
    session_ids = [row[0] for row in cursor.fetchall()]
    
    # 使用 FineTuningFormatter 生成 JSONL
    from utils.finetuning_formatter import FineTuningFormatter
    formatter = FineTuningFormatter()
    
    expert_samples = []
    scammer_samples = []
    
    for session_id in session_ids:
        # 獲取對話歷史
        cursor.execute("""
            SELECT speaker, message, round_number
            FROM conversation_messages
            WHERE session_id = ?
            ORDER BY round_number
        """, (session_id,))
        
        messages = cursor.fetchall()
        
        # 轉換格式並生成樣本
        conversation_history = [
            {"role": msg[0], "content": msg[1]}
            for msg in messages
        ]
        
        # 生成訓練樣本
        expert_samples.extend(
            formatter.format_for_expert_training(conversation_history, {}, {})
        )
        scammer_samples.extend(
            formatter.format_for_scammer_training(conversation_history, {}, {})
        )
    
    # 保存到文件
    dataset_name = f"{request.data_type}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
    
    expert_file = formatter.save_to_jsonl(expert_samples, "expert", dataset_name)
    scammer_file = formatter.save_to_jsonl(scammer_samples, "scammer", dataset_name)
    
    # 記錄到數據集表
    cursor.execute('''
        INSERT INTO finetuning_datasets
        (dataset_name, dataset_type, source_sessions, session_count,
         total_samples, expert_samples, scammer_samples,
         expert_file_path, scammer_file_path)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    ''', (
        dataset_name,
        request.data_type,
        json.dumps(session_ids),
        len(session_ids),
        len(expert_samples) + len(scammer_samples),
        len(expert_samples),
        len(scammer_samples),
        expert_file,
        scammer_file
    ))
    
    conn.commit()
    conn.close()
    
    return {
        "success": True,
        "dataset_name": dataset_name,
        "session_count": len(session_ids),
        "expert_samples": len(expert_samples),
        "scammer_samples": len(scammer_samples),
        "files": {
            "expert": expert_file,
            "scammer": scammer_file
        }
    }
```

```python
@router.post("/api/data/curate-player-data")
async def curate_player_data(request: CuratePlayerDataRequest):
    """策展高質量玩家數據用於訓練"""
    conn = sqlite3.connect(DATABASE_PATH)
    cursor = conn.cursor()
    
    # 獲取高質量玩家會話
    query = """
        SELECT ts.id, ts.player_score, pa.timing_score, pa.quality_score
        FROM training_sessions ts
        LEFT JOIN (
            SELECT session_id,
                   AVG(timing_score) as timing_score,
                   AVG(quality_score) as quality_score
            FROM player_actions
            GROUP BY session_id
        ) pa ON ts.id = pa.session_id
        WHERE ts.session_type = 'player_mode'
        AND ts.status = 'completed'
        AND ts.player_score >= ?  -- 玩家得分門檻
        AND pa.quality_score >= ?  -- 行為質量門檻
        ORDER BY ts.player_score DESC, pa.quality_score DESC
        LIMIT ?
    """
    
    cursor.execute(query, (
        request.min_player_score or 70,
        request.min_quality_score or 60,
        request.max_sessions or 100
    ))
    
    selected_sessions = cursor.fetchall()
    
    # 分析玩家策略特徵
    strategies = analyze_player_strategies(selected_sessions)
    
    # 導出為訓練數據
    result = export_data(ExportDataRequest(
        data_type="player_mode",
        session_ids=[s[0] for s in selected_sessions]
    ))
    
    conn.close()
    
    return {
        "success": True,
        "selected_sessions": len(selected_sessions),
        "average_score": sum(s[1] for s in selected_sessions) / len(selected_sessions),
        "strategy_insights": strategies,
        "dataset": result
    }
```

```python
# backend/utils/data_quality_analyzer.py

class DataQualityAnalyzer:
    """評估玩家數據的真實性和質量"""
    
    def analyze_player_data_quality(self, session_id: str) -> Dict[str, Any]:
        """分析單個會話的數據質量"""
        conn = sqlite3.connect(DATABASE_PATH)
        cursor = conn.cursor()
        
        # 獲取會話數據
        cursor.execute("""
            SELECT * FROM training_sessions WHERE id = ?
        """, (session_id,))
        session = cursor.fetchone()
        
        # 獲取玩家行為
        cursor.execute("""
            SELECT * FROM player_actions WHERE session_id = ?
        """, (session_id,))
        actions = cursor.fetchall()
        
        quality_score = 0
        indicators = {}
        
        # 1. 時機一致性 (玩家是否在關鍵時刻介入)
        timing_consistency = self.evaluate_timing_consistency(actions)
        quality_score += timing_consistency * 0.3
        indicators['timing_consistency'] = timing_consistency
        
        # 2. 策略連貫性 (玩家策略是否前後一致)
        strategy_coherence = self.evaluate_strategy_coherence(actions)
        quality_score += strategy_coherence * 0.25
        indicators['strategy_coherence'] = strategy_coherence
        
        # 3. 響應多樣性 (避免重複或機械化回應)
        response_diversity = self.evaluate_response_diversity(actions)
        quality_score += response_diversity * 0.2
        indicators['response_diversity'] = response_diversity
        
        # 4. 上下文相關性 (回應是否與對話相關)
        context_relevance = self.evaluate_context_relevance(actions)
        quality_score += context_relevance * 0.25
        indicators['context_relevance'] = context_relevance
        
        conn.close()
        
        return {
            "session_id": session_id,
            "quality_score": quality_score,
            "indicators": indicators,
            "is_high_quality": quality_score >= 70,
            "recommended_for_training": quality_score >= 60
        }
    
    def batch_analyze_player_data(self, player_id: Optional[str] = None) -> Dict[str, Any]:
        """批量分析玩家數據質量"""
        conn = sqlite3.connect(DATABASE_PATH)
        cursor = conn.cursor()
        
        query = """
            SELECT id FROM training_sessions
            WHERE session_type = 'player_mode'
            AND status = 'completed'
        """
        
        if player_id:
            query += " AND player_id = ?"
            cursor.execute(query, (player_id,))
        else:
            cursor.execute(query)
        
        session_ids = [row[0] for row in cursor.fetchall()]
        
        results = []
        for session_id in session_ids:
            quality = self.analyze_player_data_quality(session_id)
            results.append(quality)
        
        # 統計
        high_quality_count = sum(1 for r in results if r['is_high_quality'])
        avg_quality = sum(r['quality_score'] for r in results) / len(results) if results else 0
        
        conn.close()
        
        return {
            "total_sessions": len(results),
            "high_quality_sessions": high_quality_count,
            "high_quality_rate": high_quality_count / len(results) if results else 0,
            "average_quality_score": avg_quality,
            "sessions": results
        }
```

## 📋 使用流程

### 自動訓練流程

```
1. RPG Maker 啟動自動訓練
   └─> session_type = 'auto_training'
       └─> 數據保存到 training_sessions
           └─> 標記為 AI vs AI 數據
               └─> 可用於初步訓練

2. 定期導出自動訓練數據
   └─> /api/data/export (data_type='auto_training')
       └─> 生成 auto_training_YYYYMMDD.jsonl
           └─> 用於快速迭代模型
```

### 玩家參與流程

```
1. 玩家開始遊戲
   └─> session_type = 'player_mode'
       └─> player_id = '玩家標識'
           └─> 數據保存到 training_sessions
               └─> 標記為 Human vs AI 數據

2. 玩家每次介入
   └─> role = 'player'
       └─> is_player_generated = TRUE
           └─> 記錄到 conversation_messages
               └─> 額外記錄到 player_actions

3. 會話結束後
   └─> 數據質量評估
       └─> quality_score >= 70 → 高質量數據
           └─> 策展為特殊數據集
               └─> 用於關鍵模型改進
```
