"""
數據庫遷移腳本：添加數據分類字段
"""

import sqlite3
import os

DATABASE_PATH = 'anti_fraud_game.db'

def migrate_database():
    print("開始數據庫遷移...")
    
    conn = sqlite3.connect(DATABASE_PATH)
    cursor = conn.cursor()
    
    # 1. 檢查 training_sessions 表是否存在
    cursor.execute("""
        SELECT name FROM sqlite_master 
        WHERE type='table' AND name='training_sessions'
    """)
    
    if not cursor.fetchone():
        # 創建新表
        print("創建 training_sessions 表...")
        cursor.execute('''
            CREATE TABLE training_sessions (
                id TEXT PRIMARY KEY,
                session_type TEXT NOT NULL DEFAULT 'player_mode',
                mode TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                completed_at TIMESTAMP,
                status TEXT DEFAULT 'in_progress',
                victim_persona TEXT,
                scam_tactic TEXT,
                player_id TEXT,
                player_role TEXT,
                player_score INTEGER,
                outcome TEXT,
                round_count INTEGER,
                scammer_score INTEGER,
                expert_score INTEGER,
                metadata TEXT
            )
        ''')
        
        # 遷移現有 sessions 數據
        print("遷移現有 sessions 數據...")
        cursor.execute("""
            INSERT INTO training_sessions 
            (id, session_type, created_at, status, victim_persona)
            SELECT 
                id, 
                'player_mode' as session_type,
                created_at,
                status,
                persona_type as victim_persona
            FROM sessions
            WHERE EXISTS (SELECT 1 FROM sessions)
        """)
    else:
        # 表已存在，添加缺失的列
        print("更新 training_sessions 表...")
        
        # 檢查並添加 session_type 列
        try:
            cursor.execute("ALTER TABLE training_sessions ADD COLUMN session_type TEXT DEFAULT 'player_mode'")
            print("✓ 添加 session_type 列")
        except sqlite3.OperationalError:
            print("✓ session_type 列已存在")
        
        # 檢查並添加 player_id 列
        try:
            cursor.execute("ALTER TABLE training_sessions ADD COLUMN player_id TEXT")
            print("✓ 添加 player_id 列")
        except sqlite3.OperationalError:
            print("✓ player_id 列已存在")
    
    # 2. 修改 conversations 表
    cursor.execute("""
        SELECT name FROM sqlite_master 
        WHERE type='table' AND name='conversation_messages'
    """)
    
    if not cursor.fetchone():
        print("創建 conversation_messages 表...")
        cursor.execute('''
            CREATE TABLE conversation_messages (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT NOT NULL,
                round_number INTEGER,
                speaker TEXT,
                message TEXT NOT NULL,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                is_player_generated BOOLEAN DEFAULT FALSE,
                trust_in_scammer INTEGER,
                trust_in_expert INTEGER,
                metrics TEXT,
                FOREIGN KEY (session_id) REFERENCES training_sessions(id)
            )
        ''')
        
        # 遷移現有對話數據
        print("遷移現有對話數據...")
        cursor.execute("""
            INSERT INTO conversation_messages
            (session_id, speaker, message, timestamp, is_player_generated)
            SELECT 
                session_id,
                role as speaker,
                message,
                timestamp,
                CASE WHEN role = 'player' THEN 1 ELSE 0 END as is_player_generated
            FROM conversations
            WHERE EXISTS (SELECT 1 FROM conversations)
        """)
    else:
        # 添加 is_player_generated 列
        try:
            cursor.execute("ALTER TABLE conversation_messages ADD COLUMN is_player_generated BOOLEAN DEFAULT FALSE")
            print("✓ 添加 is_player_generated 列")
        except sqlite3.OperationalError:
            print("✓ is_player_generated 列已存在")
    
    # 3. 創建索引
    print("創建索引...")
    try:
        cursor.execute("""
            CREATE INDEX IF NOT EXISTS idx_session_type 
            ON training_sessions(session_type)
        """)
        cursor.execute("""
            CREATE INDEX IF NOT EXISTS idx_player_id 
            ON training_sessions(player_id)
        """)
        cursor.execute("""
            CREATE INDEX IF NOT EXISTS idx_is_player_generated 
            ON conversation_messages(is_player_generated)
        """)
        print("✓ 索引創建完成")
    except sqlite3.OperationalError as e:
        print(f"索引可能已存在: {e}")
    
    conn.commit()
    conn.close()
    
    print("\n✅ 數據庫遷移完成！")
    print("\n可用的數據分類:")
    print("  - session_type: 'auto_training' | 'player_mode'")
    print("  - is_player_generated: TRUE | FALSE")

if __name__ == "__main__":
    migrate_database()
```

# backend/api/data_routes.py (新文件)

from fastapi import APIRouter
import sqlite3
import json

router = APIRouter(prefix="/api/data", tags=["Data Management"])

DATABASE_PATH = 'anti_fraud_game.db'

@router.get("/stats")
async def get_data_stats():
    """獲取數據統計"""
    conn = sqlite3.connect(DATABASE_PATH)
    cursor = conn.cursor()
    
    # 統計各類型數據
    cursor.execute("""
        SELECT 
            session_type,
            COUNT(*) as count,
            AVG(scammer_score) as avg_scammer,
            AVG(expert_score) as avg_expert
        FROM training_sessions
        WHERE status = 'completed'
        GROUP BY session_type
    """)
    
    results = cursor.fetchall()
    
    stats = {
        "auto_training": {
            "count": 0,
            "avg_scammer_score": 0,
            "avg_expert_score": 0
        },
        "player_mode": {
            "count": 0,
            "avg_scammer_score": 0,
            "avg_expert_score": 0
        }
    }
    
    for row in results:
        session_type = row[0]
        if session_type in stats:
            stats[session_type] = {
                "count": row[1],
                "avg_scammer_score": row[2] or 0,
                "avg_expert_score": row[3] or 0
            }
    
    conn.close()
    
    return {
        "success": True,
        "stats": stats,
        "total_sessions": sum(s["count"] for s in stats.values())
    }

@router.get("/player-data")
async def get_player_data_quality():
    """獲取玩家數據質量統計"""
    conn = sqlite3.connect(DATABASE_PATH)
    cursor = conn.cursor()
    
    cursor.execute("""
        SELECT 
            COUNT(*) as total,
            AVG(player_score) as avg_score,
            SUM(CASE WHEN player_score >= 70 THEN 1 ELSE 0 END) as high_quality
        FROM training_sessions
        WHERE session_type = 'player_mode'
        AND status = 'completed'
        AND player_score IS NOT NULL
    """)
    
    result = cursor.fetchone()
    conn.close()
    
    if result and result[0] > 0:
        return {
            "success": True,
            "total_player_sessions": result[0],
            "average_player_score": result[1],
            "high_quality_sessions": result[2],
            "high_quality_rate": result[2] / result[0] * 100
        }
    else:
        return {
            "success": True,
            "total_player_sessions": 0,
            "message": "尚無玩家數據"
        }
```

// rpg_maker_plugins/DataStats.js

/*:
 * @command showDataStats
 * @text 顯示數據統計
 * @desc 顯示自動訓練和玩家數據的統計
 */

PluginManager.registerCommand('DataStats', 'showDataStats', async () => {
    try {
        const response = await fetch('http://localhost:8000/api/data/stats');
        const data = await response.json();
        
        if (data.success) {
            const auto = data.stats.auto_training;
            const player = data.stats.player_mode;
            
            $gameMessage.add('=== 數據統計 ===');
            $gameMessage.add(`總會話數: ${data.total_sessions}`);
            $gameMessage.add('');
            $gameMessage.add('[自動訓練數據]');
            $gameMessage.add(`  會話數: ${auto.count}`);
            $gameMessage.add(`  騙徒平均分: ${auto.avg_scammer_score.toFixed(1)}`);
            $gameMessage.add(`  專家平均分: ${auto.avg_expert_score.toFixed(1)}`);
            $gameMessage.add('');
            $gameMessage.add('[玩家參與數據] ⭐');
            $gameMessage.add(`  會話數: ${player.count}`);
            $gameMessage.add(`  騙徒平均分: ${player.avg_scammer_score.toFixed(1)}`);
            $gameMessage.add(`  專家平均分: ${player.avg_expert_score.toFixed(1)}`);
        }
    } catch (error) {
        $gameMessage.add('獲取統計數據失敗');
        console.error(error);
    }
});
